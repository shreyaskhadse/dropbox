# Load the iris dataset
data(iris)

# Set the random seed for reproducibility
set.seed(123)

# Generate additional 900 observations with random values
extended_iris <- iris[sample(nrow(iris), 900, replace = TRUE), ]

# Function to introduce NA values in a vector without overlapping with other NA values
introduce_nas <- function(x, num_na) {
  nas <- sample(length(x), num_na)
  x[nas] <- NA
  x
}

# Introduce NA values in each variable without overlapping
num_na_per_var <- 10  # Number of NA values to introduce per variable

extended_iris$Sepal.Length <- introduce_nas(extended_iris$Sepal.Length, num_na_per_var)
extended_iris$Sepal.Width <- introduce_nas(extended_iris$Sepal.Width, num_na_per_var)
extended_iris$Petal.Length <- introduce_nas(extended_iris$Petal.Length, num_na_per_var)
extended_iris$Petal.Width <- introduce_nas(extended_iris$Petal.Width, num_na_per_var)

# Generate outliers for a specific variable
outlier_indices <- sample(nrow(extended_iris), 20)
extended_iris$Petal.Length[outlier_indices] <- extended_iris$Petal.Length[outlier_indices] + 10

# Combine the extended dataset with the original iris dataset
combined_iris <- rbind(iris, extended_iris)

# Verify the dimensions and check for NA values
dim(combined_iris)  # Should be 1000 observations and 5 variables
sum(is.na(combined_iris))  # Count the number of NA values
