# Load the iris dataset
data(iris)

# Set the random seed for reproducibility
set.seed(123)

# Generate additional 900 observations with random values
extended_iris <- iris[sample(nrow(iris), 900, replace = TRUE), ]

# Introduce NA values in specific variables randomly
na_indices <- sample(nrow(extended_iris), 50)
num_na <- 5  # Number of NA values to introduce per variable

for (i in na_indices) {
  na_vars <- sample(names(extended_iris), num_na)
  extended_iris[i, na_vars] <- NA
}

# Generate outliers for a specific variable
outlier_indices <- sample(nrow(extended_iris), 20)
extended_iris$Petal.Length[outlier_indices] <- extended_iris$Petal.Length[outlier_indices] + 10

# Combine the extended dataset with the original iris dataset
combined_iris <- rbind(iris, extended_iris)

# Verify the dimensions and check for NA values
dim(combined_iris)  # Should be 1000 observations and 5 variables
sum(is.na(combined_iris))  # Count the number of NA values
